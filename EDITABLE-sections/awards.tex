
\newcommand{\singlebulletliststyle}[1]{
  \ifthenelse{\equal{#1}{}}{}{
    \readlist\mylist{#1}%
    \begin{itemize} 
      \setlength{\itemsep}{-0.5em} % Adjust the value to reduce or increase the space
      \foreachitem\z\in\mylist[]{
        \item \z
      }
    \end{itemize}
    \bulletpadding
  }
}

\newcommand*\CVAwardsField[1][]{%
  \begingroup
  \setkeys{cvexp}{#1}
  \color{bodytxt}
  \rightminipageonly{
  \begin{minipage}[t]{\segmenttitlesareavalue\textwidth}
    \singlebulletliststyle{\bulletlist}
  \end{minipage}%
    %\hfill%
  \begin{minipage}[t]{\datesareavalue\textwidth}
    \RaggedLeft{\datesstyle{\dates}}
  \end{minipage}%
  }
  \endgroup
}
